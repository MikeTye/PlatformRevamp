type OnboardingStatus = 'not_started' | 'in_progress' | 'completed' | 'skipped';

type OnboardingResponseData = {
    onboardingStatus?: OnboardingStatus;
    onboardingStep?: number;
    onboardingSelectedRoles?: string[];
};

const API_BASE_URL = import.meta.env.VITE_API_BASE_URL ?? 'http://localhost:3000';

// Protected Route Wrapper
function ProtectedRoute({
    children,
    requireOnboardingComplete = true,
}: {
    children: React.ReactNode;
    requireOnboardingComplete?: boolean;
}) {
    const { isAuthenticated, isAuthLoading } = useAuth();
    const location = useLocation();

    const [isOnboardingLoading, setIsOnboardingLoading] = useState(true);
    const [shouldResumeOnboarding, setShouldResumeOnboarding] = useState(false);

    useEffect(() => {
        let cancelled = false;

        const checkOnboarding = async () => {
            if (isAuthLoading) return;

            if (!isAuthenticated) {
                setIsOnboardingLoading(false);
                setShouldResumeOnboarding(false);
                return;
            }

            // Let onboarding route itself load without bouncing.
            if (!requireOnboardingComplete) {
                setIsOnboardingLoading(false);
                setShouldResumeOnboarding(false);
                return;
            }

            try {
                setIsOnboardingLoading(true);

                const resp = await fetch(`${API_BASE_URL}/user-profiles/me/onboarding`, {
                    method: 'GET',
                    credentials: 'include',
                });

                if (!resp.ok) {
                    // Fail open so app is still usable if onboarding endpoint is temporarily unavailable.
                    if (!cancelled) {
                        setShouldResumeOnboarding(false);
                    }
                    return;
                }

                const json = await resp.json();
                const onboarding = json?.data as OnboardingResponseData | undefined;

                const mustResume =
                    onboarding?.onboardingStatus === 'in_progress' ||
                    (
                        onboarding?.onboardingStatus !== 'completed' &&
                        onboarding?.onboardingStatus !== 'skipped' &&
                        (
                            (Array.isArray(onboarding?.onboardingSelectedRoles) &&
                                onboarding.onboardingSelectedRoles.length > 0) ||
                            (typeof onboarding?.onboardingStep === 'number' &&
                                onboarding.onboardingStep > 0)
                        )
                    );

                if (!cancelled) {
                    setShouldResumeOnboarding(mustResume);

                    if (mustResume) {
                        const onboardingEmail =
                            sessionStorage.getItem('tce_onboarding_email')?.trim().toLowerCase();

                        if (!onboardingEmail) {
                            // optional fallback marker so OnboardingPage has a stable key prefix
                            sessionStorage.setItem('tce_onboarding_email', 'authenticated-user');
                        }
                    }
                }
            } catch {
                if (!cancelled) {
                    setShouldResumeOnboarding(false);
                }
            } finally {
                if (!cancelled) {
                    setIsOnboardingLoading(false);
                }
            }
        };

        void checkOnboarding();

        return () => {
            cancelled = true;
        };
    }, [isAuthenticated, isAuthLoading, requireOnboardingComplete]);

    if (isAuthLoading || isOnboardingLoading) {
        return null; // or replace with a loading spinner if preferred
    }

    if (!isAuthenticated) {
        return (
            <Navigate
                to="/login"
                state={{ from: location }}
                replace
            />
        );
    }

    if (
        requireOnboardingComplete &&
        shouldResumeOnboarding &&
        location.pathname !== '/onboarding'
    ) {
        return <Navigate to="/onboarding" replace />;
    }

    return <>{children}</>;
}

// Layout Wrapper to conditionally render AppLayout
function LayoutWrapper({ children }: { children: React.ReactNode; }) {
    return <AppLayout>{children}</AppLayout>;
}
