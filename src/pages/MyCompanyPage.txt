// Sidebar Editor Component
interface SidebarEditorProps {
  open: boolean;
  onClose: () => void;
  section: string;
  editingItem?: any;
  onSave?: (section: string, data: any, isEditing: boolean) => void;
}
function SidebarEditor({
  open,
  onClose,
  section,
  editingItem,
  onSave
}: SidebarEditorProps) {
  const [sectionVisibility, setSectionVisibility] = useState(true);
  // Form states - initialized empty, will be set by useEffect
  const [name, setName] = useState('');
  const [roles, setRoles] = useState<string[]>(['Project Developer']);
  const [website, setWebsite] = useState('');
  const [shortDesc, setShortDesc] = useState('');
  const [fullDesc, setFullDesc] = useState('');
  const [country, setCountry] = useState('Malaysia');
  const [mediaCaption, setMediaCaption] = useState('');
  const [selectedRegions, setSelectedRegions] = useState<string[]>([
  'Southeast Asia',
  'South Asia']
  );
  const [selectedProjectTypes, setSelectedProjectTypes] = useState<string[]>([
  'redd',
  'peatland']
  );
  const [selectedServiceCategories, setSelectedServiceCategories] = useState<
    string[]>(
    ['MRV', 'Community engagement', 'Project development']);
  const [docName, setDocName] = useState('');
  const [docType, setDocType] = useState('');
  const [docFile, setDocFile] = useState<File | null>(null);
  // Team member states
  const [addMode, setAddMode] = useState<'search' | 'invite'>('search');
  const [selectedUser, setSelectedUser] = useState<
    (typeof existingUsers)[0] | null>(
    null);
  const [inviteEmail, setInviteEmail] = useState('');
  const [inviteName, setInviteName] = useState('');
  const [teamRole, setTeamRole] = useState('');
  const [teamOtherRole, setTeamOtherRole] = useState('');
  // Permissions state
  const [permissions, setPermissions] =
  useState<MemberPermission[]>(initialPermissions);
  const [inheritToProjects, setInheritToProjects] = useState(false);
  const [permSelectedUser, setPermSelectedUser] = useState<
    (typeof existingUsers)[0] | null>(
    null);
  const [permNewRole, setPermNewRole] = useState<PermissionLevel>('Viewer');
  // Reset/populate form when opening or when editingItem changes
  useEffect(() => {
    if (!open) return;
    if (section === 'documents') {
      if (editingItem) {
        // Editing existing document - populate with data
        setDocName(editingItem.name || '');
        setDocType(editingItem.type || '');
      } else {
        // Adding new document - clear form
        setDocName('');
        setDocType('');
      }
    } else if (section === 'team') {
      if (editingItem) {
        // Editing existing team member - populate with data
        const isStandardRole =
        teamRoles.includes(editingItem.role) && editingItem.role !== 'Other';
        if (isStandardRole) {
          setTeamRole(editingItem.role || '');
          setTeamOtherRole('');
        } else {
          // Non-standard role means it was entered as "Other"
          setTeamRole('Other');
          setTeamOtherRole(editingItem.role || '');
        }
        setInviteName(editingItem.name || '');
        // Find matching user if exists
        const matchingUser = existingUsers.find(
          (u) => u.name === editingItem.name
        );
        if (matchingUser) {
          setSelectedUser(matchingUser);
          setAddMode('search');
        } else {
          setSelectedUser(null);
          setAddMode('invite');
        }
        setInviteEmail('');
      } else {
        // Adding new team member - clear form
        setTeamRole('');
        setTeamOtherRole('');
        setSelectedUser(null);
        setInviteName('');
        setInviteEmail('');
        setAddMode('search');
      }
    } else if (section === 'media') {
      if (editingItem) {
        setMediaCaption(editingItem.caption || '');
      } else {
        setMediaCaption('');
      }
    } else if (section === 'header') {
      // Header always shows company data (not add/edit pattern)
      setName('Borneo Carbon Partners');
      setRoles(['Project Developer', 'Service Provider']);
      setWebsite('borneocarbon.com');
      setShortDesc(
        'Leading peatland restoration and conservation developer in Malaysian Borneo.'
      );
    } else if (section === 'about') {
      setFullDesc(
        'Borneo Carbon Partners is a pioneering carbon project developer focused on peatland restoration and rainforest conservation in Malaysian Borneo. Founded in 2018, we work closely with local communities and government agencies to develop high-integrity carbon projects that deliver measurable climate and biodiversity benefits.'
      );
    }
  }, [open, section, editingItem]);
  const renderVisibilityToggle = () =>
  <Box
    sx={{
      p: 2,
      bgcolor: 'grey.50',
      borderRadius: 1,
      mb: 3
    }}>
    
      <FormControlLabel
      control={
      <Switch
        checked={sectionVisibility}
        onChange={(e) => setSectionVisibility(e.target.checked)} />

      }
      label={
      <Box display="flex" alignItems="center" gap={1}>
            {sectionVisibility ?
        <VisibilityRounded
          sx={{
            fontSize: 16
          }} /> :


        <VisibilityOffRounded
          sx={{
            fontSize: 16
          }} />

        }
            <Typography variant="body2">
              {sectionVisibility ? 'Public' : 'Hidden'}
            </Typography>
          </Box>
      } />
    
      <Typography
      variant="caption"
      color="text.secondary"
      display="block"
      mt={0.5}>
      
        {sectionVisibility ?
      'This section is visible to everyone' :
      'This section is hidden from public view'}
      </Typography>
    </Box>;

  const renderContent = () => {
    switch (section) {
      case 'header':
        return (
          <Stack spacing={3}>
            <Box>
              <Typography variant="subtitle1" fontWeight="bold" gutterBottom>
                Company Details
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Basic information about your organization.
              </Typography>
            </Box>

            <Box display="flex" justifyContent="center" mb={2}>
              <Box
                sx={{
                  width: 100,
                  height: 100,
                  borderRadius: 2,
                  border: '2px dashed',
                  borderColor: 'grey.300',
                  display: 'flex',
                  flexDirection: 'column',
                  alignItems: 'center',
                  justifyContent: 'center',
                  cursor: 'pointer',
                  bgcolor: 'grey.50',
                  '&:hover': {
                    borderColor: 'grey.400',
                    bgcolor: 'grey.100'
                  }
                }}>
                
                <ImageRounded
                  sx={{
                    fontSize: 32,
                    color: 'grey.400',
                    mb: 0.5
                  }} />
                
                <Typography variant="caption" color="text.secondary">
                  Upload Logo
                </Typography>
              </Box>
            </Box>

            <TextField
              label="Company Name"
              fullWidth
              value={name}
              onChange={(e) => setName(e.target.value)} />
            

            <Box>
              <Typography variant="body2" fontWeight={500} mb={1}>
                Company Role(s)
              </Typography>
              <Typography
                variant="caption"
                color="text.secondary"
                display="block"
                mb={1.5}>
                
                Select all that apply.
              </Typography>
              <Box display="flex" gap={1.5} flexDirection="column">
                {['Project Developer', 'Service Provider'].map((role) =>
                <Card
                  key={role}
                  variant="outlined"
                  sx={{
                    borderColor: roles.includes(role) ?
                    'primary.main' :
                    'grey.200',
                    borderWidth: roles.includes(role) ? 2 : 1,
                    bgcolor: roles.includes(role) ?
                    'primary.50' :
                    'transparent'
                  }}>
                  
                    <CardActionArea
                    onClick={() => {
                      setRoles((prev) =>
                      prev.includes(role) ?
                      prev.filter((r) => r !== role) :
                      [...prev, role]
                      );
                    }}
                    sx={{
                      p: 2
                    }}>
                    
                      <Box display="flex" alignItems="flex-start" gap={1.5}>
                        {roles.includes(role) ?
                      <CheckCircleRounded color="primary" /> :

                      <RadioButtonUncheckedRounded
                        sx={{
                          color: 'grey.400'
                        }} />

                      }
                        <Box>
                          <Typography variant="subtitle2" fontWeight="bold">
                            {role}
                          </Typography>
                          <Typography variant="caption" color="text.secondary">
                            {role === 'Project Developer' ?
                          'Develops and manages carbon projects' :
                          'Provides services to carbon projects'}
                          </Typography>
                        </Box>
                      </Box>
                    </CardActionArea>
                  </Card>
                )}
              </Box>
            </Box>

            <TextField
              label="Website"
              fullWidth
              value={website}
              onChange={(e) => setWebsite(e.target.value)}
              InputProps={{
                startAdornment:
                <InputAdornment position="start">
                    <LanguageRounded
                    sx={{
                      fontSize: 16,
                      color: 'grey.400'
                    }} />
                  
                  </InputAdornment>

              }} />
            

            <TextField
              label="Short Description"
              fullWidth
              multiline
              minRows={2}
              value={shortDesc}
              onChange={(e) => setShortDesc(e.target.value)}
              placeholder="Brief summary..." />
            
          </Stack>);

      case 'about':
        return (
          <Stack spacing={3}>
            <Box>
              <Typography variant="subtitle1" fontWeight="bold" gutterBottom>
                About
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Tell your company's story.
              </Typography>
            </Box>
            {renderVisibilityToggle()}
            <TextField
              label="Full Description"
              fullWidth
              multiline
              minRows={6}
              value={fullDesc}
              onChange={(e) => setFullDesc(e.target.value)}
              placeholder="Detailed description of your company..." />
            
          </Stack>);

      case 'media':
        return (
          <Stack spacing={3}>
            <Box>
              <Typography variant="subtitle1" fontWeight="bold" gutterBottom>
                {editingItem ? 'Edit Media' : 'Add Media'}
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Upload photos and videos.
              </Typography>
            </Box>
            {renderVisibilityToggle()}
            {!editingItem &&
            <Paper
              variant="outlined"
              sx={{
                p: 3,
                borderStyle: 'dashed',
                textAlign: 'center',
                cursor: 'pointer'
              }}
              component="label">
              
                <input type="file" hidden accept="image/*,video/*" />
                <ImageRounded
                sx={{
                  fontSize: 32,
                  color: 'grey.400',
                  mb: 1
                }} />
              
                <Typography variant="body2" color="text.secondary">
                  Click to upload or drag and drop
                </Typography>
                <Typography variant="caption" color="text.disabled">
                  PNG, JPG, MP4 up to 50MB
                </Typography>
              </Paper>
            }
            <TextField
              label="Caption"
              fullWidth
              value={mediaCaption}
              onChange={(e) => setMediaCaption(e.target.value)}
              placeholder="Describe this media..." />
            
          </Stack>);

      case 'team':
        return (
          <Stack spacing={3}>
            <Box>
              <Typography variant="subtitle1" fontWeight="bold" gutterBottom>

    return 'Add';
  };
  return (
    <SidebarPanel
      open={open}
      onClose={onClose}
      title={getTitle()}
      onSave={handleSave}
      saveLabel={section === 'team' && !editingItem ? 'Add Member' : 'Save'}>
      
      {renderContent()}
    </SidebarPanel>);

                    actionLabel="Add Description"
                    onAction={() => openEditor('about')} />

                  }
                </Box>
              </Paper>

              {/* Media Section */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader
                  title="Media"
                  onAdd={() => openEditor('media')} />
                
                <Box p={3}>
                  <MediaGallery
                    items={company.media || []}
                    mode="grid"
                    isOwner={true}
                    onAdd={() => openEditor('media')}
                    onMenuClick={(e, item, index) =>
                    handleMediaMenuOpen(e, item, index)
                    }
                    emptyStateMessage="No media yet. Add photos and videos to showcase your work." />
                  
                </Box>
              </Paper>

              {/* Projects Section */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader
                  title="Projects"
                  count={company.projects?.length}
                  onAdd={() => setProjectWizardOpen(true)} />
                
                <Box p={3}>
                  {company.projects && company.projects.length > 0 ?
                  <Stack spacing={1.5}>
                      {company.projects.map((project) =>
                    <ProjectCard
                      key={project.upid}
                      upid={project.upid}
                      name={project.name}
                      developer={company.name}
                      stage={project.stage}
                      type={project.type}
                      country={project.country}
                      countryCode={project.countryCode}
                      hectares={project.hectares}
                      expectedCredits={project.expectedCredits}
                      onClick={() => navigate(`/projects/${project.upid}`)}
                      variant="compact" />

                    )}
                    </Stack> :

                  <EmptyState
                    icon={FolderRounded}
                    title="No projects yet"
                    description="Create your first carbon project to get started"
                    actionLabel="Create Project"
                    onAction={() => setProjectWizardOpen(true)} />

                  }
                </Box>
              </Paper>

              {/* Services Section - shown when company has Service Provider role */}
              {company.roles?.includes('Service Provider') &&
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                  <SectionHeader
                  title="Services Offered"
                  count={company.services?.length}
                  onEdit={() => openEditor('services')} />
                
                  <Box p={3}>
                    {company.services && company.services.length > 0 ?
                  <Box display="flex" flexDirection="column" gap={1.5}>
                        {company.serviceCategories &&
                    company.serviceCategories.length > 0 &&
                    <Box display="flex" flexWrap="wrap" gap={1} mb={1}>
                              {company.serviceCategories.map((cat) =>
                      <Chip
                        key={cat}
                        label={cat}
                        size="small"
                        sx={{
                          height: 24,
                          fontSize: '0.75rem',
                          bgcolor: 'grey.100',
                          color: 'grey.700',
                          fontWeight: 500
                        }} />

                      )}
                            </Box>
                    }
                        <Box
                      display="flex"
                      flexDirection={{
                        xs: 'column',
                        sm: 'row'
                      }}
                      flexWrap="wrap"
                      gap={1.5}>
                      
                          {company.services.map((service) =>
                      <Box
                        key={service}
                        p={1.5}
                        bgcolor="grey.50"
                        border={1}
                        borderColor="grey.100"
                        borderRadius={1}
                        sx={{
                          flex: {
                            xs: '1 1 100%',
                            sm: '1 1 calc(50% - 6px)'
                          },
                          minWidth: {
                            xs: '100%',
                            sm: 'calc(50% - 6px)'
                          }
                        }}>
                        
                              <Typography variant="body2" color="text.primary">
                                {service}
                              </Typography>
                            </Box>
                      )}
                        </Box>
                      </Box> :

                  <EmptyState
                    icon={GroupRounded}
                    title="No services listed"
                    description="Add the services your company offers"
                    actionLabel="Add Services"
                    onAction={() => openEditor('services')} />

                  }
                  </Box>
                </Paper>
              }

              {/* Documents Section */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader
                  title="Documents"
                  onAdd={() => openEditor('documents')} />
                
                <Box p={3}>
                  {company.documents && company.documents.length > 0 ?
                  <Stack spacing={1}>
                      {company.documents.map((doc, i) =>
                    <Box
                      key={i}
                      display="flex"
                      alignItems="center"
                      gap={1.5}
                      p={1}
                      borderRadius={1}
                      sx={{
                        '&:hover': {
                          bgcolor: 'grey.50'
                        }
                      }}>
                      
                          <DescriptionRounded
                        sx={{
                          fontSize: 16,
                          color: 'grey.500'
                        }} />
                      
                          <Box minWidth={0} flex={1}>
                            <Typography
                          variant="body2"
                          fontWeight="medium"
                          color="text.primary"
                          noWrap>
                          
                              {doc.name}
                            </Typography>
                            <Typography
                          variant="caption"
                          color="text.secondary">
                          
                              {doc.type} • {doc.date}
                            </Typography>
                          </Box>
                          <IconButton
                        size="small"
                        onClick={() => {}}
                        sx={{
                          color: 'grey.400',
                          '&:hover': {
                            color: 'primary.main'
                          }
                        }}
                        title="Download">
                        
                            <DownloadRounded
                          sx={{
                            fontSize: 16
                          }} />
                        
                          </IconButton>
                          <IconButton
                        size="small"
                        onClick={(e) => handleDocumentMenuOpen(e, doc, i)}
                        sx={{
                          color: 'grey.400',
                          '&:hover': {
                            color: 'grey.600'
                          }
                        }}>
                        
                            <MoreVertRounded
                          sx={{
                            fontSize: 18
                          }} />
                        
                          </IconButton>
                        </Box>
                    )}
                    </Stack> :

                  <EmptyState
                    icon={DescriptionRounded}
                    title="No documents yet"
                    description="Upload company documents for due diligence"
                    actionLabel="Upload Document"
                    onAction={() => openEditor('documents')} />

                  }
                </Box>
              </Paper>
            </Stack>
          </Box>

          {/* Sidebar - 1/3 width */}
          <Box
            sx={{
              flex: {
                xs: '1 1 100%',
                md: '0 0 33.333%'
              },
              maxWidth: {
                xs: '100%',
                md: '33.333%'
              }
            }}>
            
            <Stack spacing={2}>
              {/* Profile Completeness */}
              {completenessItems.filter((i) => i.isComplete).length <
              completenessItems.length &&
              <ProfileCompleteness
                title="Improve Listing"
                items={completenessItems}
                onItemClick={handleCompletenessItemClick} />

              }

              {/* Team */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader title="Team" onAdd={() => openEditor('team')} />
                <Box p={2}>
                  {company.team && company.team.length > 0 ?
                  <Stack spacing={1.5}>
                      {company.team.map((member, i) => {
                      const isInvited = member.invited;
                      const userSlug = member.name.
                      toLowerCase().
                      replace(/\s+/g, '-');
                      return (
                        <Box
                          key={member.name}
                          display="flex"
                          alignItems="center"
                          gap={1.5}>
                          
                            <Avatar
                            onClick={
                            isInvited ?
                            undefined :
                            () => navigate(`/users/${userSlug}`)
                            }
                            sx={{
                              width: 32,
                              height: 32,
                              bgcolor: isInvited ? 'grey.200' : 'grey.100',
                              color: isInvited ? 'grey.400' : 'grey.500',
                              fontSize: '0.75rem',
                              fontWeight: 500,
                              cursor: isInvited ? 'default' : 'pointer',
                              border: isInvited ? '1px dashed' : 'none',
                              borderColor: 'grey.300'
                            }}>
                            
                              {member.name.
                            split(' ').
                            map((n) => n[0]).
                            join('')}
                            </Avatar>
                            <Box minWidth={0} flex={1}>
                              <Box
                              display="flex"
                              alignItems="center"
                              gap={0.75}>
                              
                                <Typography
                                variant="body2"
                                fontWeight="medium"
                                color={
                                isInvited ?
                                'text.secondary' :
                                'primary.main'
                                }
                                onClick={
                                isInvited ?
                                undefined :
                                () => navigate(`/users/${userSlug}`)
                                }
                                sx={
                                isInvited ?
                                {} :
                                {
                                  cursor: 'pointer',
                                  '&:hover': {
                                    textDecoration: 'underline'
                                  }
                                }
                                }>
                                
                                  {member.name}
                                </Typography>

                                {isInvited &&
                              <Chip
                                label="Invited"
                                size="small"
                                sx={{
                                  height: 20,
                                  fontSize: '0.65rem',
                                  fontWeight: 600,
                                  bgcolor: 'transparent',
                                  color: 'warning.dark',
                                  border: '1px solid',
                                  borderColor: 'warning.200'
                                }} />

                              }
                              </Box>
                              <Typography
                              variant="caption"
                              color="text.secondary">
                              
                                {member.role}
                              </Typography>
                            </Box>
                            <IconButton
                            size="small"
                            onClick={(e) => handleTeamMenuOpen(e, member, i)}
                            sx={{
                              color: 'grey.400',
                              '&:hover': {
                                color: 'grey.600'
                              }
                            }}>
                            
                              <MoreVertRounded
                              sx={{
                                fontSize: 18
                              }} />
                            
                            </IconButton>
                          </Box>);

                    })}
                    </Stack> :

                  <Box
                    onClick={() => openEditor('team')}
                    sx={{
                      p: 2,
                      border: '1px dashed',
                      borderColor: 'grey.300',
                      borderRadius: 1,
                      bgcolor: 'grey.50',
                      textAlign: 'center',
                      cursor: 'pointer',
                      '&:hover': {
                        bgcolor: 'grey.100'
                      }
                    }}>
                    
                      <GroupRounded
                      sx={{
                        fontSize: 20,
                        color: 'grey.400',
                        mb: 0.5
                      }} />
                    
                      <Typography
                      variant="caption"
                      color="text.disabled"
                      display="block">
                      
                        Add team members
                      </Typography>
                    </Box>
                  }
                </Box>
              </Paper>

              {/* Location */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader
                  title="Location"
                  onEdit={() => openEditor('location')} />
                
                <Box p={2}>
                  <Box display="flex" alignItems="center" gap={1}>
                    <Typography fontSize="1.125rem">
                      {FLAGS[company.countryCode] || '🌍'}
                    </Typography>
                    <Typography variant="body2" color="text.primary">
                      {company.country}
                    </Typography>
                  </Box>
                </Box>
              </Paper>

              {/* Geographical Coverage */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader
                  title="Geographical Coverage"
                  onEdit={() => openEditor('coverage')} />
                
                <Box p={2}>
                  {company.geographicalCoverage &&
                  company.geographicalCoverage.length > 0 ?
                  <Box display="flex" flexWrap="wrap" gap={1}>
                      {company.geographicalCoverage.map((country) =>
                    <Chip
                      key={country}
                      label={country}
                      size="small"
                      sx={{
                        bgcolor: 'grey.100',
                        color: 'grey.700'
                      }} />

                    )}
                    </Box> :

                  <Typography
                    variant="body2"
                    color="text.disabled"
                    fontStyle="italic"
                    onClick={() => openEditor('coverage')}
                    sx={{
                      cursor: 'pointer'
                    }}>
                    
                      Add regions where you operate
                    </Typography>
                  }
                </Box>
              </Paper>

              {/* Permissions */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader
                  title="Permissions"
                  onEdit={() => openEditor('permissions')} />
                
                <Box p={2}>
                  <Stack spacing={1}>
                    {initialPermissions.slice(0, 3).map((member) =>
                    <Box
                      key={member.id}
                      display="flex"
                      alignItems="center"
                      gap={1.5}>
                      
                        <Avatar
                        sx={{
                          width: 28,
                          height: 28,
                          bgcolor: 'grey.100',
                          color: 'grey.600',
                          fontSize: '0.65rem',
                          fontWeight: 600
                        }}>
                        
                          {member.name.
                        split(' ').
                        map((n) => n[0]).
                        join('')}
                        </Avatar>
                        <Box flex={1} minWidth={0}>
                          <Typography variant="body2" fontWeight={500} noWrap>
                            {member.name}
                          </Typography>
                        </Box>
                        <Chip
                        label={member.permission}
                        size="small"
                        sx={{
                          height: 20,
                          fontSize: '0.65rem',
                          fontWeight: 600,
                          bgcolor:
                          member.permission === 'Owner' ?
                          'grey.800' :
                          member.permission === 'Admin' ?
                          'grey.200' :
                          'grey.100',
                          color:
                          member.permission === 'Owner' ?
                          'white' :
                          'grey.700'
                        }} />
                      
                      </Box>
                    )}
                    <Button
                      size="small"
                      onClick={() => openEditor('permissions')}
                      sx={{
                        textTransform: 'none',
                        color: 'text.secondary',
                        justifyContent: 'flex-start',
                        pl: 0,
                        mt: 0.5
                      }}>
                      
                      Manage permissions →
                    </Button>
                  </Stack>
                </Box>
              </Paper>

              {/* Capabilities */}
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                <SectionHeader
                  title="Project Types"
                  onEdit={() => openEditor('projectTypes')} />
                
                <Box p={2}>
                  {company.projectTypes && company.projectTypes.length > 0 ?
                  <Box display="flex" flexWrap="wrap" gap={1}>
                      {company.projectTypes.map((type) =>
                    <Chip
                      key={type}
                      label={type.toUpperCase()}
                      size="small"
                      variant="outlined"
                      sx={{
                        height: 24,
                        fontSize: '0.625rem'
                      }} />

                    )}
                    </Box> :

                  <Typography
                    variant="body2"
                    color="text.disabled"
                    fontStyle="italic"
                    onClick={() => openEditor('projectTypes')}
                    sx={{
                      cursor: 'pointer'
                    }}>
                    
                      Add project types you work with
                    </Typography>
                  }
                </Box>
              </Paper>

              {/* Service Categories - shown when company has Service Provider role */}
              {company.roles?.includes('Service Provider') &&
              <Paper
                variant="outlined"
                sx={{
                  borderRadius: 2,
                  overflow: 'hidden'
                }}>
                
                  <SectionHeader
                  title="Service Categories"
                  onEdit={() => openEditor('services')} />
                
                  <Box p={2}>
                    {company.serviceCategories &&
                  company.serviceCategories.length > 0 ?
                  <Box display="flex" flexWrap="wrap" gap={1}>
                        {company.serviceCategories.map((cat) =>
                    <Chip
                      key={cat}
                      label={cat}
                      size="small"
                      sx={{
                        height: 24,
                        fontSize: '0.75rem',
                        bgcolor: 'grey.100',
                        color: 'grey.700',
                        fontWeight: 500
                      }} />

                    )}
                      </Box> :

                  <Typography
                    variant="body2"
                    color="text.disabled"
                    fontStyle="italic"
                    onClick={() => openEditor('services')}
                    sx={{
                      cursor: 'pointer'
                    }}>
                    
                        Add service categories
                      </Typography>
                  }
                  </Box>
                </Paper>
              }
            </Stack>
          </Box>
        </Box>
      </Box>

      {/* Sidebar Editor */}
      <SidebarEditor
        open={sidebarOpen}
        onClose={() => {
          setSidebarOpen(false);
          setEditingItem(null);
        }}
        section={sidebarSection}
        editingItem={editingItem}
        onSave={handleSidebarSave} />
      

      {/* Project Wizard */}
      <ProjectWizard
        open={projectWizardOpen}
        onClose={() => setProjectWizardOpen(false)}
        hasCompanies={true} />
      

      {/* Document Menu */}
      <Menu
        anchorEl={documentMenuAnchor}
        open={Boolean(documentMenuAnchor)}
        onClose={handleDocumentMenuClose}
        anchorOrigin={{
          vertical: 'bottom',
          horizontal: 'right'
        }}
        transformOrigin={{
          vertical: 'top',
          horizontal: 'right'
        }}
        PaperProps={{
          sx: {
            minWidth: 160,
            boxShadow: 3
          }
        }}>
        
        <MenuItem onClick={handleEditDocument}>
          <ListItemIcon>
            <EditRounded
              sx={{
                fontSize: 18
              }} />
            
          </ListItemIcon>
          <ListItemText
            primary="Edit"
            primaryTypographyProps={{
              variant: 'body2'
            }} />
          
        </MenuItem>
        <MenuItem
          onClick={handleDeleteDocument}
          sx={{
            color: 'error.main'
          }}>
          
          <ListItemIcon>
            <DeleteRounded
              sx={{
                fontSize: 18,
                color: 'error.main'
              }} />
            
          </ListItemIcon>
          <ListItemText
            primary="Delete"
            primaryTypographyProps={{
              variant: 'body2'
            }} />
          
        </MenuItem>
      </Menu>

      {/* Team Member Menu */}
      <Menu
        anchorEl={teamMenuAnchor}
        open={Boolean(teamMenuAnchor)}
        onClose={handleTeamMenuClose}
        anchorOrigin={{
          vertical: 'bottom',
          horizontal: 'right'
        }}
        transformOrigin={{
          vertical: 'top',
          horizontal: 'right'
        }}
        PaperProps={{
          sx: {
            minWidth: 160,
            boxShadow: 3
          }
        }}>
        
        <MenuItem onClick={handleEditTeamMember}>
          <ListItemIcon>
            <EditRounded
              sx={{
                fontSize: 18
              }} />
            
          </ListItemIcon>
          <ListItemText
            primary="Edit"
            primaryTypographyProps={{
              variant: 'body2'
            }} />
          
        </MenuItem>
        <MenuItem
          onClick={handleDeleteTeamMember}
          sx={{
            color: 'error.main'
          }}>
          
          <ListItemIcon>
            <DeleteRounded
              sx={{
                fontSize: 18,
                color: 'error.main'
              }} />
            
          </ListItemIcon>
          <ListItemText
            primary="Remove"
            primaryTypographyProps={{
              variant: 'body2'
            }} />
          
        </MenuItem>
      </Menu>

      {/* Media Menu */}
      <Menu
        anchorEl={mediaMenuAnchor}
        open={Boolean(mediaMenuAnchor)}
        onClose={handleMediaMenuClose}
        anchorOrigin={{
          vertical: 'bottom',
          horizontal: 'right'
        }}
        transformOrigin={{
          vertical: 'top',
          horizontal: 'right'
        }}
        PaperProps={{
          sx: {
            minWidth: 160,
            boxShadow: 3
          }
        }}>
        
        <MenuItem onClick={handleEditMedia}>
          <ListItemIcon>
            <EditRounded
              sx={{
                fontSize: 18
              }} />
            
          </ListItemIcon>
          <ListItemText
            primary="Edit"
            primaryTypographyProps={{
              variant: 'body2'
            }} />
          
        </MenuItem>
        <MenuItem
          onClick={handleDeleteMedia}
          sx={{
            color: 'error.main'
          }}>
          
          <ListItemIcon>
            <DeleteRounded
              sx={{
                fontSize: 18,
                color: 'error.main'
              }} />
            
          </ListItemIcon>
          <ListItemText
            primary="Delete"
            primaryTypographyProps={{
              variant: 'body2'
            }} />
          
        </MenuItem>
      </Menu>

      {/* Delete Confirmation Dialog */}
      <Dialog
        open={deleteDialogOpen}
        onClose={handleDeleteDialogClose}
        PaperProps={{
          sx: {
            borderRadius: 2,
            maxWidth: 400
          }
        }}>
        
        <DialogTitle
          sx={{
            pb: 1
          }}>
          
          <Typography variant="h6" fontWeight="bold">
            Confirm Delete
          </Typography>
        </DialogTitle>
        <DialogContent>
          <DialogContentText>
            Are you sure you want to delete{' '}
            <strong>{itemToDelete?.name || 'this item'}</strong>? This action
            can be undone within 15 seconds.
          </DialogContentText>
        </DialogContent>
        <DialogActions
          sx={{
            px: 3,
            pb: 2
          }}>
          
          <Button
            onClick={handleDeleteDialogClose}
            sx={{
              color: 'text.secondary',
              textTransform: 'none'
            }}>
            
            Cancel
          </Button>
          <Button
            onClick={handleConfirmDelete}
            variant="contained"
            color="error"
            sx={{
              textTransform: 'none'
            }}>
            
            Delete
          </Button>
        </DialogActions>
      </Dialog>

      {/* Snackbar with Undo */}
      <Snackbar
        open={snackbarOpen}
        autoHideDuration={15000}
        onClose={handleSnackbarClose}
        anchorOrigin={{
          vertical: 'bottom',
          horizontal: 'center'
        }}>
        
        <Alert
          onClose={() => setSnackbarOpen(false)}
          severity="success"
          variant="filled"
          sx={{
            width: '100%',
            alignItems: 'center',
            '& .MuiAlert-message': {
              display: 'flex',
              alignItems: 'center',
              gap: 2
            }
          }}
          action={
          deletedItem ?
          <Button
            color="inherit"
            size="small"
            onClick={handleUndo}
            sx={{
              textTransform: 'none',
              fontWeight: 'bold',
              ml: 1
            }}>
            
                Undo
              </Button> :
          null
          }>
          
          {snackbarMessage}
        </Alert>
      </Snackbar>
    </Box>);

}
